import React from "react";
import { Grid, GridColumn as Column } from "@progress/kendo-react-grid";

const GridList = ({ gridData, getData }) => {
  const handleColorClass = (value) => {
    if (value === 1) {
      return "greenCell";
    } else if (value === 2) {
      return "yellowCell";
    } else if (value === 3) {
      return "orangeCell";
    } else if (value === 4) {
      return "redCell";
    }
    return "";
  };

  return (
    <Grid
      data={gridData.data}
      {...gridData.initialDataState}
      onItemChange={getData}
    >
      {gridData.columns.map((column) => (
        <Column
          key={column.field}
          field={column.field}
          title={column.title}
          cell={(props) => (
            <td className={handleColorClass(props.dataItem.JudgementWeightage)}>
              {props.dataItem[column.field]}
            </td>
          )}
        />
      ))}
    </Grid>
  );
};

export default GridList;
